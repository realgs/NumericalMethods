import time
data = [9, 5, 7, 6, 6, 4, 3, 8, 1, 2]
start = time.clock()
def insert_sort(data):
    for i in range(1,len(data)):
        insert = data[i]
        j = i - 1
        while j>=0 and data[j]>insert:
            data[j + 1] = data[j]
            j = j - 1
        data[j + 1] = insert
        
insert_sort(data)
end = time.clock()
total = end - start
print(data)
print("{0:02f}s".format(total))
